<?php
    $a = 5;
    $b = '05';
    var_dump($a == $b);         // Почему true? По тому, что после преобразования типов данных b становится равно 5.
    var_dump((int)'012345');     // Почему 12345? Тип int не содержит 0.
    var_dump((float)123.0 === (int)123.0); // Почему false? По тому, что === сравнивает типы данных и здесь они разные.
    var_dump((int)0 === (int)'hello, world'); // Почему true? По тому, что int обнулит данные и оба значения станут 0.
?>
